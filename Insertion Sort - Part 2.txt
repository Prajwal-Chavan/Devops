void insertionSort2(int n, int arr_count, int* arr) {
    for(int i=1; i < n; i++){
        int ele = arr[i];
        int j = i-1;
        while (j >= 0 && arr[j] > ele) {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = ele;
        
        for(int k=0;k<n;k++){
            printf("%d ", arr[k]);
        }
        printf("\n");
    }


}
